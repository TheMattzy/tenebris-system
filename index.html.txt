    // Check factory password
    function checkFactoryPassword() {
      const input = document.getElementById('factory-password');
      const password = input.value;
      const levelCombos = factoryPasswords[currentFactoryLevel];

      const normalize = (str) => {
        let normalized = str.toLowerCase();
        normalized = normalized.normalize('NFD').replace(/[\u0300-\u036f]/g, '');
        return normalized;
      };

      const normalizedPassword = normalize(password);
      let isValid = false;

      for (let combo of levelCombos) {
        let matches = false;
        if (selectedFactoryEmoji === '💎') {
          matches = (
            combo.emoji === '💎' &&
            combo.color === selectedGemColor &&
            normalize(combo.password) === normalizedPassword
          );
        } else {
          matches = (
            combo.emoji === selectedFactoryEmoji &&
            combo.color === null &&
            normalize(combo.password) === normalizedPassword
          );
        }
        if (matches) {
          isValid = true;
          break;
        }
      }

      if (!isValid) {
        factoryErrors.push({
          level: currentFactoryLevel,
          errorCode: `ERR_${Math.random().toString(36).substr(2, 6).toUpperCase()}`
        });
      }

      currentFactoryLevel++;
      selectedFactoryEmoji = null;
      selectedGemColor = null;
      factoryStep = 'emoji';
      input.value = '';

      if (currentFactoryLevel > 13) {
        closeModal();
        if (factoryErrors.length > 0) {
          setTimeout(() => {
            const errorModal = document.getElementById('modal');
            const errorContent = document.getElementById('modal-content');
            const firstError = factoryErrors[0];
            errorContent.innerHTML = `
              <div class="text-center">
                <div class="relative mb-4">
                  <p class="text-6xl text-red-500">✕</p>
                </div>
                <p class="text-3xl text-red-500 font-bold mb-2">ACESSO NEGADO</p>
                <p class="text-2xl text-red-600 mb-4">FALHAS DETECTADAS</p>
                <p class="text-sm text-red-400 mb-6 font-mono">[ ${firstError.errorCode} ]</p>
                <p class="text-sm text-red-400 mb-6">Foram detectadas ${factoryErrors.length} falha(s) na sequência de segurança.</p>
                <button onclick="resetFactoryAfterError()" class="px-6 py-3 border-2 border-red-500 text-red-500 hover:bg-red-500 hover:text-black transition-all">
                  [ACEITAR]
                </button>
              </div>
            `;
            errorModal.classList.remove('hidden');
            errorModal.classList.add('flex');
            errorModal.style.animation = 'none';
            errorModal.offsetHeight;
            errorModal.style.animation = 'emergency-alert 0.4s ease-in-out 8';
            playEmergencySound();
            setTimeout(() => {
              errorModal.style.animation = 'none';
            }, 3200);
          }, 300);
        } else {
          setTimeout(() => {
            const successModal = document.getElementById('modal');
            const successContent = document.getElementById('modal-content');
            successContent.innerHTML = `
              <div class="text-center">
                <div class="relative mb-4">
                  <p class="text-6xl text-green-500 glitch">✓✓</p>
                </div>
                <p class="text-4xl text-green-500 font-bold mb-2 glitch">ACESSO LIBERADO</p>
                <p class="text-sm text-green-400 mb-6">Todos os níveis de segurança ultrapassados</p>
                <button onclick="unlockFactorySuccess()" class="px-6 py-3 border-2 border-green-500 bg-green-500 text-black font-bold hover:bg-green-600 transition-all">

        renderFactoryChallenge(document.getElementById('modal-content'));

    // Play emergency sound
    function playEmergencySound() {
      try {
        const audioContext = new (window.AudioContext || window.webkitAudioContext)();
        const oscillator = audioContext.createOscillator();
        const gainNode = audioContext.createGain();
        oscillator.connect(gainNode);
        gainNode.connect(audioContext.destination);

        oscillator.frequency.value = 900;
        gainNode.gain.setValueAtTime(0.3, audioContext.currentTime);

        for (let i = 0; i < 8; i++) {
          const startTime = audioContext.currentTime + (i * 0.3);
          gainNode.gain.setValueAtTime(0.3, startTime);
          gainNode.gain.setValueAtTime(0, startTime + 0.15);
        }
        oscillator.start(audioContext.currentTime);
        oscillator.stop(audioContext.currentTime + 2.4);
      } catch (e) {
        console.error('Audio context not available');
      }
    }
